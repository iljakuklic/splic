fn power(x: [[u64]], exp: u64) -> [[u64]] {
    match exp {
        0 => #(1),
        1 => x,
        exp => {
            let exp2: [[u64]] = #{
                let x2 = $(power(x, exp / 2));
                x2 * x2
            };
            let odd: u64 = exp & 1;
            match odd == 1 {
                0 => exp2,
                1 => #($(exp2) * $(x)),
            }
        }
    }
}

code fn pow5(x: u64) -> u64 {
    $(power(#(x), 5))
}
